    95→    95→export async function buildQuotePrompt(ctx: AgentContext): Promise<{ system: string; user: string }> {
    96→    96→  const tenant = await db.tenant.findUnique({ where: { id: ctx.tenantId } })
    97→    97→  if (!tenant) throw new Error('Tenant not found')
    98→    98→  const system = `Eres el motor de cotización de ${tenant.slug}. Recibes uno o más SKU de
    99→    99→interés y la cantidad de cada uno. Consulta precios_por_volumen (filtrado
   100→   100→por tenant_id) por cada SKU según tipo_cliente=${ctx.perfil || 'detal'} y cantidad. Suma
   101→   101→el total a pagar, la venta estimada usando precio_ref_mercado, y el
   102→   102→margen total. Responde en el formato: "[cantidad] [producto] + [cantidad]
   103→   103→[producto]: pagas $[total] → vendes $[venta] → te sobran $[margen]
   104→   104→limpios". Nunca inventes un precio que no exista en la tabla. Si el SKU
   105→   105→no existe para este tenant, dilo explícitamente.`
   106→   106→  // Fetch real volume prices for the items
   107→   107→  const itemsInfo: string[] = []
   108→   108→  if (ctx.items && ctx.items.length > 0) {
   109→   109→    for (const it of ctx.items) {
   110→   110→      const product = await db.product.findUnique({ where: { tenantId_sku: { tenantId: ctx.tenantId, sku: it.sku } } })
   111→   111→      if (product) {
   112→   112→        const vols = await db.volumePrice.findMany({ where: { tenantId: ctx.tenantId, productId: product.id, tipoCliente: ctx.perfil || 'detal' }, orderBy: { cantidadMinima: 'asc' }})
   113→   113→        const vol = vols.find(v => it.cantidad >= v.cantidadMinima && it.cantidad <= v.cantidadMaxima)
   114→   114→        itemsInfo.push(`SKU ${it.sku} (${product.name}) × ${it.cantidad}: precio base ${product.price}, ${vol ? `precio volumen ${vol.precioUnitario} (tramo ${vol.cantidadMinima}-${vol.cantidadMaxima})` : 'sin tramo de volumen aplicable'}`)
   115→   115→      } else {
   116→   116→        itemsInfo.push(`SKU ${it.sku}: NO EXISTE en el catálogo de este tenant`)
   117→   117→      }
   118→   118→    }
   119→   119→  }
   120→   120→  const user = `Items a cotizar:
   121→   121→${itemsInfo.join('\n') || 'Sin items especificados — pide al cliente qué quiere cotizar.'}`
   122→   122→  return { system, user }
   123→   123→}

   128→   128→export async function buildCatalogPrompt(ctx: AgentContext): Promise<{ system: string; user: string }> {
   129→   129→  const tenant = await db.tenant.findUnique({ where: { id: ctx.tenantId } })
   130→   130→  if (!tenant) throw new Error('Tenant not found')
   131→   131→  const system = `Eres el agente de catálogo de ${tenant.slug}. Cuando el lead pregunta por un
   132→   132→producto, tema o categoría, tu respuesta NUNCA puede ser solo texto ni un
   133→   133→enlace genérico. Busca en el catálogo real de este tenant (embedding_texto,
   134→   134→sincronizado desde WhatsApp Catalog, WooCommerce, Shopify, o la Supabase
   135→   135→propia o nuestra) el producto o los productos que mejor coinciden con la
   136→   136→intención del lead, y devuelve sus imágenes reales. Si la intención agrupa
   137→   137→una categoría amplia (ej. "familia"), trae mínimo 3 prendas distintas
   138→   138→disponibles en esa categoría (consulta combos_categoria), no solo el
   139→   139→producto ancla. Acompaña con un máximo de 1-2 líneas de texto. Cierra
   140→   140→siempre con una pregunta binaria, nunca una pregunta abierta.`
   141→   141→  // Search the catalog
   142→   142→  const q = ctx.query || ''
   143→   143→  let products: { sku: string; name: string; imageUrl: string | null; price: number; diseno: string | null; categoria: string | null }[] = []
   144→   144→  if (q) {
   145→   145→    const combo = await db.categoryCombo.findUnique({ where: { tenantId_categoria: { tenantId: ctx.tenantId, categoria: q.toLowerCase() } } })
   146→   146→    if (combo) {
   147→   147→      const skus = combo.skusRecomendados.split(',')
   148→   148→      products = await db.product.findMany({ where: { tenantId: ctx.tenantId, sku: { in: skus }, active: true }})
   149→   149→    } else {
   150→   150→      products = await db.product.findMany({ where: { tenantId: ctx.tenantId, OR: [{ name: { contains: q } }, { diseno: { contains: q } }, { categoria: { contains: q } }, { description: { contains: q } }], active: true }, take: 5 })
   151→   151→    }
   152→   152→  } else {
   153→   153→    products = await db.product.findMany({ where: { tenantId: ctx.tenantId, active: true }, take: 5 })
   154→   154→  }
   155→   155→  const user = `Catálogo disponible (filtrado por "${q}"):
   156→   156→${products.map(p => `- ${p.sku}: ${p.name} ($${p.price}) [${p.diseno || 'liso'}] img: ${p.imageUrl || 'sin imagen'}`).join('\n') || 'Sin productos encontrados — pregunta al cliente qué busca.'}`
   157→   157→  return { system, user }
   158→   158→}

   829→   829→  const system = `Eres el analista de direcciones de ${tenant.slug} (Colombia-focused). Antes de despachar,
   830→   830→evalúas la dirección del cliente contra: (1) completitud de los 10 campos Saramantha, (2) coberturas
   831→   831→de las transportadoras configuradas para este tenant, (3) historial de entrega de esa dirección
   832→   832→para este tenant (¿hubo rechazo, novedad, devolución?), (4) normalización (barrio, vía, número,
   833→   833→interior, referencias). Salida JSON:
   834→   834→{"direccion_completa": bool, "campos_faltantes": [...], "cobertura": "nacional|internacional|sin_cobertura",
   835→   835→"transportadoras_disponibles": [...], "riesgo_entrega": "bajo|medio|alto", "historial_previo": "ok|rechazo|novedad|sin_registro",
   836→   836→"accion_recomendada": "despachar|confirmar_direccion|pedir_referencia|rechazar_envio",
   837→   837→"pregunta_cliente": "..."}. Nunca inventes un resultado de entrega que no esté en el historial.`
   838→   838→  let history = ''
   839→   839→  let partialAddress = ctx.partialAddress || {}
   840→   840→  if (ctx.customerId) {
   841→   841→    const h = await db.deliveryHistory.findMany({ where: { tenantId: ctx.tenantId, contactoId: ctx.customerId }, take: 5 })
   842→   842→    history = h.map(d => `- ${d.direccionNormalizada}, ${d.ciudad}: ${d.resultadoEntregaAnterior || 'sin registro'}`).join('\n')
   843→   843→  }
   844→   844→  if (ctx.orderId && Object.keys(partialAddress).length === 0) {
   845→   845→    const order = await db.order.findUnique({ where: { id: ctx.orderId }, select: { address: true, city: true, country: true } })
   846→   846→    if (order) partialAddress = { direccion: order.address || '', ciudad: order.city || '', pais: order.country || 'CO' }
   847→   847→  }
   848→   848→  const carriers = await db.carrier.findMany({ where: { tenantId: ctx.tenantId }, select: { nombreCanonico: true, cobertura: true } })
   849→   849→  const user = `Proveedor logístico principal: ${tenant.proveedorLogistico}
   850→   850→Transportadoras canónicas: ${carriers.map(c => `${c.nombreCanonico} (${c.cobertura})`).join(', ') || 'ninguna configurada'}
   851→   851→
   852→   852→Dirección a analizar: ${JSON.stringify(partialAddress)}
   853→   853→
   854→   854→Historial de entrega de este contacto:
   855→   855→${history || 'Sin historial previo.'}
   856→   856→
   857→   857→Mensaje del cliente: "${ctx.message || '(sin mensaje — solo análisis estructural)'}"`
   858→   858→  return { system, user }
   859→   859→}
