Verificación HMAC (WA)

// src/lib/middleware/hmac.ts
import { timingSafeEqual } from 'crypto'

export function verifyMetaSignature(
  rawBody: string,
  signature: string | null,
  appSecret: string
): boolean {
  if (!signature) return false
  const expected = 'sha256=' +
    hmacSha256(rawBody, appSecret)
  const a = Buffer.from(signature)
  const b = Buffer.from(expected)
  return a.length === b.length &&
    timingSafeEqual(a, b)
}

// Con WA_APP_SECRET set + signature válida

// scripts/test-chat.ts
import { io } from 'socket.io-client'

const s = io('http://localhost:3003', {
  auth: { token: process.env.TEST_TOKEN },
  transports: ['websocket']
})

// 1. connect
s.on('connect', () => {
  console.log('✓ connected', s.id)
  s.emit('room:join', { roomId: 'ten-saramantha' })

  console.log('✓ room joined', r))

  console.log('✓ auth rejected', e.message))

Fix 1 — next.config allowedDevOrigins

⚠️ Síntoma
En dev, el browser lanzaba warnings de Cross-Origin check failed cuando el HMR intentaba conectar desde un host distinto a localhost. No rompía funcionalidad, pero generaba ruido en consola.
✅ Fix
Añadida config allowedDevOrigins en next.config.ts con los hosts del entorno de QA. Warnings eliminados, HMR funciona limpio.
// next.config.ts
const nextConfig = {
  reactStrictMode: true,
  ignoreBuildErrors: false,
  allowedDevOrigins: [
    'http://localhost:3000',
    'http://127.0.0.1:3000',
    'http://commerceflow.local:3000',
  ],
  // ...resto
}
