    23→export interface AgentContext {

    25→  conversationId?: string

    29→  // For vision agent: incoming image URL

    44→export async function buildProfilePrompt(ctx: AgentContext): Promise<{ system: string; user: string }> {
    45→  const tenant = await db.tenant.findUnique({ where: { id: ctx.tenantId } })
    46→  if (!tenant) throw new Error('Tenant not found')
    47→  const system = `Eres el clasificador de perfil del negocio ${tenant.slug}. Tu única tarea es
    48→determinar el perfil del lead a partir de su mensaje y el contexto del
    49→anuncio que lo trajo: mayorista (tienda/surtir/vender/negocio), emprendedor
    50→(arrancar/emprender), detal (para mí) o regalo. Si no hay señal clara,
    51→responde exactamente la pregunta_perfil configurada para este tenant y no
    52→avances hasta recibir respuesta. Nunca preguntes el perfil antes de haber
    53→recibido y procesado la imagen o video inicial del anuncio, si lo hay.
    54→Responde solo con el perfil detectado o la pregunta — nada más.`
    55→  const user = `Pregunta_perfil configurada para este tenant: "${tenant.preguntaPerfil || '¿Para ti o para tu negocio?'}"`
    56→  return { system, user }
    57→}

    62→export async function buildSpeechPrompt(ctx: AgentContext): Promise<{ system: string; user: string }> {
    63→  const tenant = await db.tenant.findUnique({ where: { id: ctx.tenantId } })
    64→  if (!tenant) throw new Error('Tenant not found')
    65→  const speech = ctx.perfil ? await db.salesSpeech.findUnique({ where: { tenantId_perfil: { tenantId: ctx.tenantId, perfil: ctx.perfil } } }) : null
    66→  const system = `Eres la asesora de ventas de ${tenant.slug} (nombre_asesora configurado en
    67→clientes_plataforma). Tuteas, con certeza total, sin disculpas. Cada
    68→mensaje cierra con una acción. El perfil del lead ya fue determinado:
    69→${ctx.perfil || 'pendiente'}. Consulta discursos_por_perfil para este tenant y ese perfil, y
    70→usa su apertura_texto y prueba_social tal como están, adaptando solo el
    71→tono configurado (tono_marca). No inventes datos de la empresa que no
    72→estén en la tabla clientes_plataforma o contactos. Máximo 20 palabras por
    73→mensaje, máximo 2 emojis, nunca preguntas abiertas después de dar el precio.`
    74→  const user = `Nombre_asesora: ${tenant.nombreAsesora || 'Asesora'}
    75→Tono_marca: ${tenant.tonoMarca || 'Cercano, profesional'}
    76→${speech ? `Apertura para perfil ${ctx.perfil}: ${speech.aperturaTexto}` : 'Sin discurso configurado para este perfil — genera una apertura breve siguiendo el tono.'}
    77→${speech?.pruebaSocial ? `Prueba social: ${speech.pruebaSocial}` : ''}`
    78→  return { system, user }
    79→}

    82→// 6.3 — Agente de ofertas y cotización cruzada
    83→// ────────────────────────────────────────────────────────────────────
    84→export async function buildQuotePrompt(ctx: AgentContext): Promise<{ system: string; user: string }> {
    85→  const tenant = await db.tenant.findUnique({ where: { id: ctx.tenantId } })
    86→  if (!tenant) throw new Error('Tenant not found')
    87→  const system = `Eres el motor de cotización de ${tenant.slug}. Recibes uno o más SKU de
    88→interés y la cantidad de cada uno. Consulta precios_por_volumen (filtrado
    89→por tenant_id) por cada SKU según tipo_cliente=${ctx.perfil || 'detal'} y cantidad. Suma
    90→el total a pagar, la venta estimada usando precio_ref_mercado, y el
    91→margen total. Responde en el formato: "[cantidad] [producto] + [cantidad]
    92→[producto]: pagas $[total] → vendes $[venta] → te sobran $[margen]
    93→limpios". Nunca inventes un precio que no exista en la tabla. Si el SKU
    94→no existe para este tenant, dilo explícitamente.`
    95→  // Fetch real volume prices for the items
    96→  const itemsInfo: string[] = []
    97→  if (ctx.items && ctx.items.length > 0) {
    98→    for (const it of ctx.items) {
    99→      const product = await db.product.findUnique({ where: { tenantId_sku: { tenantId: ctx.tenantId, sku: it.sku } } })
   100→      if (product) {
   101→        const vols = await db.volumePrice.findMany({ where: { tenantId: ctx.tenantId, productId: product.id, tipoCliente: ctx.perfil || 'detal' }, orderBy: { cantidadMinima: 'asc' }})
   102→        const vol = vols.find(v => it.cantidad >= v.cantidadMinima && it.cantidad <= v.cantidadMaxima)
   103→        itemsInfo.push(`SKU ${it.sku} (${product.name}) × ${it.cantidad}: precio base ${product.price}, ${vol ? `precio volumen ${vol.precioUnitario} (tramo ${vol.cantidadMinima}-${vol.cantidadMaxima})` : 'sin tramo de volumen aplicable'}`)
   104→      } else {
   105→        itemsInfo.push(`SKU ${it.sku}: NO EXISTE en el catálogo de este tenant`)
   106→      }
   107→    }
   108→  }
   109→  const user = `Items a cotizar:
   110→${itemsInfo.join('\n') || 'Sin items especificados — pide al cliente qué quiere cotizar.'}`
   111→  return { system, user }
   112→}

   117→export async function buildCatalogPrompt(ctx: AgentContext): Promise<{ system: string; user: string }> {
   118→  const tenant = await db.tenant.findUnique({ where: { id: ctx.tenantId } })
   119→  if (!tenant) throw new Error('Tenant not found')
   120→  const system = `Eres el agente de catálogo de ${tenant.slug}. Cuando el lead pregunta por un
   121→producto, tema o categoría, tu respuesta NUNCA puede ser solo texto ni un
   122→enlace genérico. Busca en el catálogo real de este tenant (embedding_texto,
   123→sincronizado desde WhatsApp Catalog, WooCommerce, Shopify, o la Supabase
   124→propia o nuestra) el producto o los productos que mejor coinciden con la
   125→intención del lead, y devuelve sus imágenes reales. Si la intención agrupa
   126→una categoría amplia (ej. "familia"), trae mínimo 3 prendas distintas
   127→disponibles en esa categoría (consulta combos_categoria), no solo el
   128→producto ancla. Acompaña con un máximo de 1-2 líneas de texto. Cierra
   129→siempre con una pregunta binaria, nunca una pregunta abierta.`
   130→  // Search the catalog
   131→  const q = ctx.query || ''
   132→  let products: { sku: string; name: string; imageUrl: string | null; price: number; diseno: string | null; categoria: string | null }[] = []
   133→  if (q) {
   134→    const combo = await db.categoryCombo.findUnique({ where: { tenantId_categoria: { tenantId: ctx.tenantId, categoria: q.toLowerCase() } } })
   135→    if (combo) {
   136→      const skus = combo.skusRecomendados.split(',')
   137→      products = await db.product.findMany({ where: { tenantId: ctx.tenantId, sku: { in: skus }, active: true }})
   138→    } else {
   139→      products = await db.product.findMany({ where: { tenantId: ctx.tenantId, OR: [{ name: { contains: q } }, { diseno: { contains: q } }, { categoria: { contains: q } }, { description: { contains: q } }], active: true }, take: 5 })
   140→    }
   141→  } else {
   142→    products = await db.product.findMany({ where: { tenantId: ctx.tenantId, active: true }, take: 5 })
   143→  }
   144→  const user = `Catálogo disponible (filtrado por "${q}"):
   145→${products.map(p => `- ${p.sku}: ${p.name} ($${p.price}) [${p.diseno || 'liso'}] img: ${p.imageUrl || 'sin imagen'}`).join('\n') || 'Sin productos encontrados — pregunta al cliente qué busca.'}`
   146→  return { system, user }
   147→}

   190→export async function buildAddressPrompt(ctx: AgentContext): Promise<{ system: string; user: string }> {
   191→  const tenant = await db.tenant.findUnique({ where: { id: ctx.tenantId } })
   192→  if (!tenant) throw new Error('Tenant not found')
   193→  const country = ctx.country || ctx.partialAddress?.pais || 'CO'
   194→  const system = `Eres el agente de datos de ${tenant.slug}. Cuando el lead confirma que quiere
   195→comprar, extrae de la conversación: nombre, apellido, teléfono,
   196→departamento, ciudad, dirección, horario, talla, diseño y cantidad.
   197→Pregunta solo los campos que falten, uno a la vez si es necesario. Al
   198→completar los 10 campos, normaliza la dirección y consulta
   199→historial_entrega_direccion (filtrado por tenant_id) antes de confirmar
   200→el pedido.
   201→
   202→VALIDACIÓN DE CALIDAD DE DIRECCIÓN (país: ${country}):
   203→Después de recolectar la dirección, VALIDA su calidad antes de confirmar:
   204→- Si la dirección incluye tipo de vía + número + referencia → "dirección correcta"
   205→- Si falta información clave → pide lo que falte: "falta que proporcione [datos]"
   206→- Para direcciones rurales: vereda + finca = correcta
   207→- Para oficinas: nombre + complemento = correcta
   208→- Si la dirección termina en guión, ignóralo (no indica incompleta)
   209→- Para oficinas de transportadoras: solo despachamos a Interrapidísimo o Coordinadora
   210→
   211→Esta validación se hace EN EL MISMO PASO de recolección — no es un agente separado.
   212→Responde "dirección correcta" o "falta que proporcione [datos]" después de recolectar.`
   213→  // Fetch delivery history for this customer
   214→  let history = ''
   215→  if (ctx.customerId) {
   216→    const h = await db.deliveryHistory.findMany({ where: { tenantId: ctx.tenantId, contactoId: ctx.customerId }, take: 5 })
   217→    history = h.map(d => `- ${d.direccionNormalizada}, ${d.ciudad}: ${d.resultadoEntregaAnterior || 'sin registro'}`).join('\n')
   218→  }
   219→  const user = `País: ${country}
   220→Datos parciales ya extraídos: ${JSON.stringify(ctx.partialAddress || {})}
   221→
   222→Historial de entrega de este contacto:
   223→${history || 'Sin historial previo.'}`
   224→  return { system, user }
   225→}

   232→  if (!tenant) throw new Error('Tenant not found')

   252→export async function buildVisionPrompt(ctx: AgentContext): Promise<{ system: string; user: string }> {
   253→  const tenant = await db.tenant.findUnique({ where: { id: ctx.tenantId } })
   254→  if (!tenant) throw new Error('Tenant not found')
   255→  const system = `Eres el agente de visión de ${tenant.slug}. Identificas productos del catálogo real a partir de imágenes enviadas por el cliente.
   256→
   257→Reglas estrictas:
   258→1. La franja de metadata visible en cada imagen del catálogo contiene SKU, diseño y precio de referencia.
   259→2. Tu PRIORIDAD es leer esa franja y devolver el SKU exacto. NO inventes.
   260→3. Si la franja está recortada o ilegible, compara visualmente contra los productos del catálogo de este tenant y devuelve el SKU más probable con tu confianza (0-1).
   261→4. Si la confianza es baja (< 0.6), responde pidiendo al cliente que confirme el diseño, sin asumir cuál es.
   262→5. Responde SOLO en formato JSON: {"sku": "...", "confianza": 0.0-1.0, "metodo": "ocr_franja|comparacion_visual|sin_match", "pregunta_confirmacion": "..." | null}`
   263→  // Fetch catalog for visual comparison
   264→  const products = await db.product.findMany({ where: { tenantId: ctx.tenantId, active: true }, take: 20, select: { sku: true, name: true, diseno: true, price: true, imageUrl: true }})
   265→  const user = `Catálogo de referencia (compara visualmente contra estos):
   266→${products.map(p => `- ${p.sku}: ${p.name} [${p.diseno || 'liso'}] $${p.price} img: ${p.imageUrl || 'sin imagen'}`).join('\n')}
   267→
   268→Imagen del cliente: ${ctx.imageUrl || 'NO PROPORCIONADA — pide al cliente que envíe la foto del producto'}`
   269→  return { system, user }
   270→}

   274→// ────────────────────────────────────────────────────────────────────
   275→export async function buildCheckoutPrompt(ctx: AgentContext): Promise<{ system: string; user: string }> {
   276→  const tenant = await db.tenant.findUnique({ where: { id: ctx.tenantId } })
   277→  if (!tenant) throw new Error('Tenant not found')
   278→  const system = `Eres el agente de checkout de ${tenant.slug}. Cuando el pedido está confirmado (datos completos, flete cotizado), preparas el resumen final para el cliente y disparas el proceso de checkout:
   279→
   280→1. Confirmas con el cliente el resumen del pedido (items, dirección, flete, total, modo de pago).
   281→2. Si pago anticipado: generas el link del carrito y lo envías.
   282→3. Si contra entrega: confirmas que el pago se hará al recibir.
   283→4. Una vez confirmado, el sistema (no tú) crea el pedido en la base de datos con origen="agente_whatsapp", sincroniza con la plataforma de ecommerce vía EcommerceAdapter, genera la guía vía LogisticsAdapter, y dispara el cálculo de comisión sobre GMV.
   284→
   285→Tu mensaje al cliente debe ser el resumen + una pregunta binaria de confirmación final. Máximo 30 palabras + lista de items.`
   286→  const user = `Política de pago: ${tenant.politicaPago}
   287→Modo de pago preferido para este pedido: (definido por el agente de discurso/contexto)
   288→
   289→(Los datos reales del pedido — items, dirección, flete, total — se pasan en el contexto de la conversación.)`
   290→  return { system, user }
   291→}

   302→    case 'theme': return buildThemePrompt(ctx)
   303→    case 'objection': return buildObjectionPrompt(ctx)
   304→    case 'address': return buildAddressPrompt(ctx)
   305→    case 'logistics': return buildLogisticsPrompt(ctx)
   306→    case 'vision': return buildVisionPrompt(ctx)
   307→    case 'checkout': return buildCheckoutPrompt(ctx)
   308→    case 'guide_tracking': return buildGuideTrackingPrompt(ctx)
   309→    case 'customer_score': return buildCustomerScorePrompt(ctx)
   310→    case 'carrier_score': return buildCarrierScorePrompt(ctx)
   311→    case 'guide_alert': return buildGuideAlertPrompt(ctx)
   312→    case 'marketplace': return buildMarketplacePrompt(ctx)
   313→    case 'remarketing': return buildRemarketingPrompt(ctx)
   314→    case 'address_analysis': return buildAddressAnalysisPrompt(ctx)
   315→    case 'novedades': return buildNovedadesPrompt(ctx)
   316→    case 'redelivery': return buildRedeliveryPrompt(ctx)
   317→    case 'buyer_behavior': return buildBuyerBehaviorPrompt(ctx)
   318→    case 'cart_builder': return buildCartBuilderPrompt(ctx)
   319→    case 'product_enrichment': return buildProductEnrichmentPrompt(ctx)
   320→    case 'affiliator': return buildAffiliatorPrompt(ctx)
   321→    case 'traffic_orchestrator': return buildTrafficOrchestratorPrompt(ctx)
   322→    case 'sales_retainer': return buildSalesRetainerPrompt(ctx)
   323→    case 'logistics_notifier': return buildLogisticsNotifierPrompt(ctx)
   324→    default: throw new Error(`Unknown agent: ${agentName}`)
   325→  }
   326→}
