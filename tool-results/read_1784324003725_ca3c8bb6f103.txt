   512→<script>
   513→let current=0;
   514→const slides=document.querySelectorAll('.slide');
   515→const total=slides.length;
   516→const counter=document.getElementById('counter');
   517→const prevBtn=document.getElementById('prev');
   518→const nextBtn=document.getElementById('next');
   519→const progress=document.getElementById('progress');
   520→
   521→function show(n){
   522→  slides[current].classList.remove('active');
   523→  current=Math.max(0,Math.min(total-1,n));
   524→  slides[current].classList.add('active');
   525→  counter.textContent=(current+1)+' / '+total;
   526→  prevBtn.disabled=current===0;
   527→  nextBtn.disabled=current===total-1;
   528→  progress.style.width=((current+1)/total*100)+'%';
   529→}
   530→
   531→function navigate(dir){show(current+dir);}
   532→
   533→document.addEventListener('keydown',e=>{
   534→  if(e.key==='ArrowRight'||e.key===' '||e.key==='PageDown')navigate(1);
   535→  if(e.key==='ArrowLeft'||e.key==='PageUp')navigate(-1);
   536→  if(e.key==='Home')show(0);
   537→  if(e.key==='End')show(total-1);
   538→});
   539→
   540→let touchX=0;
   541→document.addEventListener('touchstart',e=>touchX=e.touches[0].clientX);
   542→document.addEventListener('touchend',e=>{
   543→  const dx=e.changedTouches[0].clientX-touchX;
   544→  if(Math.abs(dx)>50)navigate(dx<0?1:-1);
   545→});
   546→
   547→show(0);
   548→</script>
