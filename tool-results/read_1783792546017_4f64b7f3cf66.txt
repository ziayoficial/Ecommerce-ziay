   295→// ───────────────────────────────────────────────────────────────────────────
   296→// CredentialCard — single integration credential editor
   297→// ───────────────────────────────────────────────────────────────────────────
   298→function CredentialCard({
   299→  integration,
   300→  saved,
   301→  onSavedChange,
   302→}: {
   303→  integration: CredentialIntegration
   304→  saved: Record<string, SavedField> | undefined
   305→  onSavedChange: (integrationId: string, fields: Record<string, SavedField> | undefined) => void
   306→}) {
   307→  const [expanded, setExpanded] = useState(false)
   308→  const [draft, setDraft] = useState<Record<string, string>>({})
   309→  const [revealed, setRevealed] = useState<Record<string, boolean>>({})
   310→  const [saving, setSaving] = useState(false)
   311→  const [testing, setTesting] = useState(false)
   312→
   313→  const configured = !!(saved && Object.keys(saved).length > 0)
   314→
   315→  const handleSave = async () => {
   316→    setSaving(true)
   317→    try {
   318→      // Only send fields the user actually typed (non-empty, non-masked).
   319→      const toSave: Record<string, string> = {}
   320→      for (const [k, v] of Object.entries(draft)) {
   321→        const sv = (v ?? '').trim()
   322→        if (sv && !sv.startsWith('••••')) toSave[k] = sv
   323→      }
   324→      if (Object.keys(toSave).length === 0) {
   325→        toast.info('Nada nuevo que guardar', {
   326→          description: 'Edita al menos un campo antes de guardar.',
   327→        })
   328→        return
   329→      }
   330→      const res = await fetch('/api/integrations/credentials', {
   331→        method: 'POST',
   332→        headers: { 'Content-Type': 'application/json' },
   333→        body: JSON.stringify({ integration: integration.id, fields: toSave }),
   334→      })
   335→      const data = await res.json()
   336→      if (!res.ok) throw new Error(data.error || 'Error al guardar')
   337→      onSavedChange(integration.id, data.fields || {})
   338→      setDraft({})
   339→      toast.success(`${integration.name}: credenciales guardadas`, {
   340→        description: `${Object.keys(data.fields || {}).length} campo(s) configurado(s)`,
   341→      })
   342→    } catch (e) {
   343→      const msg = e instanceof Error ? e.message : 'Error desconocido'
   344→      toast.error(`No se pudo guardar ${integration.name}`, { description: msg })
   345→    } finally {
   346→      setSaving(false)
   347→    }
   348→  }
   349→
   350→  const handleTest = async () => {
   351→    setTesting(true)
   352→    try {
   353→      // Send only the user's draft additions — the server merges with the
   354→      // stored secret values (which the client never sees in raw form).
   355→      const toSend: Record<string, string> = {}
   356→      for (const [k, v] of Object.entries(draft)) {
   357→        const sv = (v ?? '').trim()
   358→        if (sv && !sv.startsWith('••••')) toSend[k] = sv
   359→      }
   360→      const res = await fetch('/api/integrations/credentials', {
   361→        method: 'POST',
   362→        headers: { 'Content-Type': 'application/json' },
   363→        body: JSON.stringify({ integration: integration.id, fields: toSend, test: true }),
   364→      })
   365→      const data = await res.json()
   366→      if (!res.ok) throw new Error(data.error || 'Error al testar')
   367→      if (data.ok) {
   368→        toast.success(`${integration.name}: conexión OK`, { description: data.message })
   369→      } else {
   370→        toast.error(`${integration.name}: falló la prueba`, { description: data.message })
   371→      }
   372→    } catch (e) {
   373→      const msg = e instanceof Error ? e.message : 'Error desconocido'
   374→      toast.error(`No se pudo testar ${integration.name}`, { description: msg })
   375→    } finally {
   376→      setTesting(false)
   377→    }
   378→  }
   379→
   380→  const handleDelete = async () => {
   381→    if (!configured) return
   382→    if (!confirm(`¿Eliminar todas las credenciales guardadas de ${integration.name}?`)) return
   383→    try {
   384→      const res = await fetch(
   385→        `/api/integrations/credentials?integration=${encodeURIComponent(integration.id)}`,
   386→        { method: 'DELETE' },
   387→      )
   388→      const data = await res.json()
   389→      if (!res.ok) throw new Error(data.error || 'Error al eliminar')
   390→      onSavedChange(integration.id, undefined)
   391→      setDraft({})
   392→      toast.success(`${integration.name}: credenciales eliminadas`)
   393→    } catch (e) {
   394→      const msg = e instanceof Error ? e.message : 'Error desconocido'
   395→      toast.error(`No se pudo eliminar ${integration.name}`, { description: msg })
   396→    }
   397→  }
   398→
   399→  return (
   400→    <div
   401→      className={cn(
   402→        'rounded-xl border bg-card overflow-hidden transition-all',
   403→        configured ? 'border-emerald-500/30' : 'border-border',
   404→      )}
   405→    >
   406→      <button
   407→        type="button"
   408→        onClick={() => setExpanded(e => !e)}
   409→        className="w-full p-3 flex items-center gap-2 text-left hover:bg-muted/30 transition-colors"
   410→        aria-expanded={expanded}
   411→        aria-controls={`cred-panel-${integration.id}`}
   412→      >
   413→        <span className="text-xl leading-none shrink-0">{integration.emoji}</span>
   414→        <div className="flex-1 min-w-0">
   415→          <div className="font-medium text-sm truncate">{integration.name}</div>
   416→          <div className="text-[10px] text-muted-foreground truncate">{integration.description}</div>
   417→        </div>
   418→        <Badge
   419→          variant="outline"
   420→          className={cn(
   421→            'text-[9px] h-4 px-1.5 shrink-0',
   422→            configured
   423→              ? 'bg-emerald-500/10 text-emerald-700 dark:text-emerald-300'
   424→              : 'bg-slate-500/10 text-slate-600 dark:text-slate-300',
   425→          )}
   426→        >
   427→          {configured ? 'Configurado' : 'Pendiente'}
   428→        </Badge>
   429→        {expanded
   430→          ? <ChevronDown className="size-4 shrink-0 text-muted-foreground" />
   431→          : <ChevronRight className="size-4 shrink-0 text-muted-foreground" />}
   432→      </button>
   433→
   434→      {expanded && (
   435→        <div id={`cred-panel-${integration.id}`} className="border-t p-3 space-y-2.5 bg-muted/10">
   436→          {integration.fields.map(f => {
   437→            const sv = saved?.[f.name]
   438→            const isRevealed = !!revealed[f.name]
   439→            return (
   440→              <div key={f.name} className="space-y-1">
   441→                <div className="flex items-center justify-between gap-2">
   442→                  <Label
   443→                    htmlFor={`cred-${integration.id}-${f.name}`}
   444→                    className="text-xs text-muted-foreground truncate"
   445→                  >
   446→                    {f.label}
   447→                    {f.secret && (
   448→                      <span className="ml-1 text-[9px] uppercase tracking-wide text-amber-600/80">secret</span>
   449→                    )}
   450→                  </Label>
   451→                  <div className="flex items-center gap-1.5 shrink-0">
   452→                    {sv && (
   453→                      <span
   454→                        className="text-[9px] text-emerald-600 dark:text-emerald-400 font-mono truncate max-w-[120px]"
   455→                        title={sv.masked}
   456→                      >
   457→                        {sv.masked}
   458→                      </span>
   459→                    )}
   460→                    {f.secret && (
   461→                      <button
   462→                        type="button"
   463→                        onClick={() => setRevealed(s => ({ ...s, [f.name]: !s[f.name] }))}
   464→                        className="text-muted-foreground hover:text-foreground transition-colors p-0.5"
   465→                        aria-label={isRevealed ? 'Ocultar' : 'Mostrar'}
   466→                        title={isRevealed ? 'Ocultar' : 'Mostrar'}
   467→                      >
   468→                        {isRevealed ? <EyeOff className="size-3.5" /> : <Eye className="size-3.5" />}
   469→                      </button>
   470→                    )}
   471→                  </div>
   472→                </div>
   473→                <Input
   474→                  id={`cred-${integration.id}-${f.name}`}
